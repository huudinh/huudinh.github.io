Use JS Turtle , to draw n polygons, n entered by users 

          clear();
          function draw() {
            let x = prompt("Enter number of polygons: ");
            for (let i = 1; i <= x; i++) {
              for (let j = 1; j <= i + 2; j++) {
                fd(100);
                rt(360 / (i + 2));
              }
            }
          }
          draw();
