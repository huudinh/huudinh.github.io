Section 4 - Homework 10

Given this data structure, in which: 


            {
                shape: 'rect',
                x: 8,
                y: 70,
                width: 12,
                height: 40,
                  }
              
              rect means draw a rectangle with the respective 
              position (x, y) and size (width, height)
              
              {
                shape: 'square',
                x: 20,
                y: 40,
                width: 50,
                    },
              
              square means draw a square with the 
              respective position (x, y) and size (width)

Go through and execute all command. Note: To make turtle go to the initial position with initial angle, use home() statements. 


























            let rectangle = {
                shape: 'rect',
                x: 8,
                y: 70,
                width: 12,
                height: 40,
            };
            
            let square = {
                shape: 'square',
                x: 20,
                y: 40,
                width: 50,
            };
            
            penup();
            rt(90);
            fd(rectangle.x);
            lt(90);
            fd(rectangle.y);
            pendown();
            
            for(let i = 0; i < 4; i++) {
                if(i % 2 == 0) {
                    fd(rectangle.height);
                } else {
                    fd(rectangle.width);
                }
                rt(90);
            }
            
            home();
            
            penup();
            rt(90);
            fd(square.x);
            lt(90);
            fd(square.y);
            pendown();
            
            for(let i = 0; i < 4; i++) {
                fd(square.width);
                rt(90);
            }
